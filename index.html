          function solution(number){
            if (number <= 0) {
              return 0;
            } else {
              let arr = [];
              let sum = 0;
              for (let i = 0; i < number; i++) {
                if (i % 3 == 0 | i % 5 == 0) {
                arr.push(i);
                continue;
                } else {
                continue;
                }
              }
              for (let i = 0; i < arr.length; i++) {
                sum += arr[i];
              }
              return sum;
            }
          }
